
#include <stdlib.h>
#include <stdio.h>
#include <SDL.h>

#define largeur 800
#define hauteur 600

void SDL_ExitwithError(const char* message);

int main( int argc, char * argv[] )
{
    SDL_Window* window=NULL;SDL_Renderer *renderer=NULL;
    //initialisation
    if (SDL_Init(SDL_INIT_VIDEO)!=0)
    {
        SDL_ExitwithError("initialisation du sdl");
    }
    //window+rendu
    if(SDL_CreateWindowAndRenderer(largeur,hauteur,0,&window,&renderer)!=0)
    {
        SDL_ExitwithError("creation de fenetre+rendu");
    }

    SDL_bool programme= SDL_TRUE ;
    while(programme)
    {
        SDL_Event event;
        while(SDL_PollEvent(&event))
        {
            switch(event.type)
            {
            case SDL_MOUSEBUTTONDOWN:
                printf(" click %d %d \n",event.button.x,event.button.y);
                if(event.button.clicks>=2)
                {
                    printf("double click");
                }
                if(event.button.button==SDL_BUTTON_LEFT)
                {
                    printf("button gauche");
                }
                if(event.button.button==SDL_BUTTON_RIGHT)
                {
                    printf("click droit");
                }
                break;
            case SDL_KEYDOWN:
                switch(event.key.keysym.sym)
                {
                case SDLK_a:
                    printf("yow yow");break;
                default :
                    continue;
                }
            case SDL_QUIT :
                programme = SDL_FALSE; break;
            default: break;
            }
        }
    }


    SDL_DestroyRenderer(renderer);
    SDL_DestroyWindow(window);
    SDL_Quit();
    return EXIT_SUCCESS;
}







void SDL_ExitwithError(const char* message)
{
    printf("Erreur: %s %s",message,SDL_GetError());
    SDL_Quit();
    exit(EXIT_FAILURE);
}
